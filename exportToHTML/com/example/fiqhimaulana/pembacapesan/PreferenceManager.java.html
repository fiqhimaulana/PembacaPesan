9    public class PreferenceManager {
10           SharedPreferences pref;
11           SharedPreferences.Editor editor;
12           Context _context;
13   
14           int PRIVATE_MODE = 0;
15   
16           private static final String PREF_NAME = "intro_slider-welcome";
17   
18           private static final String IS_FIRST_TIME_LAUNCH = "IsFirstTimeLaunch";
19   
20           private static final String IS_TTS_ON = "IsTTSOn";
21   
22           public PreferenceManager(Context context) {
23               this._context = context;
24               pref = _context.getSharedPreferences(PREF_NAME, PRIVATE_MODE);
25               editor = pref.edit();
26           }
27   
28           public void setFirstTimeLaunch(boolean isFirstTime) {
29               editor.putBoolean(IS_FIRST_TIME_LAUNCH, isFirstTime);
30               editor.commit();
31           }
32   
33           public boolean isFirstTimeLaunch() {
34               return pref.getBoolean(IS_FIRST_TIME_LAUNCH, true);
35           }
36   
37           public void setIsTtsOn(boolean isTtsOn){
38               editor.putBoolean(IS_TTS_ON, isTtsOn);
39               editor.commit();
40           }
41   
42           public boolean isTtsOn(){
43               return pref.getBoolean(IS_TTS_ON, true);
44           }
45       }
